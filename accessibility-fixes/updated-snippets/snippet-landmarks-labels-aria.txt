add_action( 'wp_footer', function () {
	$js = <<<'JS'
/* Accessibility fixes (server-rendered elements exist before this footer script runs).
 * Each fix is guarded so it never overwrites an existing author value. */
(function () {
  function label(sel, text) {
    document.querySelectorAll(sel).forEach(function (el) {
      if (el && !el.getAttribute('aria-label') && !el.getAttribute('aria-labelledby')) {
        el.setAttribute('aria-label', text);
      }
    });
  }
  /* #5 landmark-unique: give each <nav> a distinct accessible name */
  label('#site-navigation', 'Primary');
  label('#mainnav', 'Primary mobile');
  label('.player-panels-nav', 'Podcast subscribe and share');

  /* #2 label-title-only: SSP player inputs carry only a title attribute. Mirror it into
   * aria-label so the accessible name is reliable (title alone is not). */
  document.querySelectorAll('.castos-player input[title], .castos-player textarea[title]').forEach(function (el) {
    if (!el.getAttribute('aria-label') && !el.getAttribute('aria-labelledby')) {
      var t = el.getAttribute('title');
      if (t) el.setAttribute('aria-label', t);
    }
  });

  /* #6 landmark-complementary-is-top-level: SSP wraps the episode meta line in a bare <aside>,
   * which implies role=complementary nested inside <main>. It is just episode metadata, not a
   * page-level aside, so drop the misplaced landmark role. Also covers the widget-sidebar
   * <aside> widgets on the Episode List page ("/podcast/"), which nest inside the #secondary
   * complementary region and were flagged the same way. */
  document.querySelectorAll('.podcast_meta aside, .podcast_player aside, #secondary aside').forEach(function (el) {
    if (!el.getAttribute('role')) el.setAttribute('role', 'presentation');
  });

  /* landmark-unique (Episode List page, "/podcast/"): SSP hardcodes the same aria-label
   * ("Podcast player" on the player region, "Podcast subscribe and share" on its nav) into
   * every player instance. That's fine when a page has one player, but the Episode List
   * template renders one full player per episode, so the labels collide. When more than one
   * element shares a landmark role + identical label, append that instance's episode title
   * (or a 1-based index as a fallback) so each is uniquely identifiable. */
  function uniquifyLandmarks(sel) {
    var groups = {};
    document.querySelectorAll(sel).forEach(function (el) {
      var lbl = el.getAttribute('aria-label') || '';
      (groups[lbl] = groups[lbl] || []).push(el);
    });
    Object.keys(groups).forEach(function (lbl) {
      var els = groups[lbl];
      if (!lbl || els.length < 2) return;
      els.forEach(function (el, i) {
        var container = el.closest('.castos-player') || el;
        var t = container.querySelector('.episode-title, .ssp-episode-title, .player__episode-title');
        var suffix = (t && t.textContent.trim()) ? t.textContent.trim() : String(i + 1);
        el.setAttribute('aria-label', lbl + ' – ' + suffix);
      });
    });
  }
  uniquifyLandmarks('[role="region"].castos-player');
  uniquifyLandmarks('.player-panels-nav');

  /* Pending #1 (WCAG 2.1.1 Keyboard): the Castos player's seek bar is a div[role="progressbar"]
   * with no tabindex — so it can never receive keyboard focus. SSP's own castos-player.min.js
   * already contains ArrowLeft/ArrowRight keydown handling for this element (dead code, since a
   * non-focusable div never fires keydown), so the actual fix is just to make it focusable with
   * the correct interactive role — SSP's existing handler does the rest. */
  document.querySelectorAll('.ssp-progress[role="progressbar"]').forEach(function (bar) {
    bar.setAttribute('role', 'slider');
    bar.setAttribute('tabindex', '0');
    if (!bar.getAttribute('aria-label') && !bar.getAttribute('aria-labelledby')) {
      bar.setAttribute('aria-label', 'Seek');
    }
  });

  /* axe 'label' violation (Cloudflare Turnstile rollout, 2026-07-07): WPForms prints a hidden
   * validation-relay text input (.wpforms-recaptcha-hidden) that its own JS populates once the
   * CAPTCHA passes, used purely to gate client-side form submission. It's visually clipped
   * (1px, clip:rect) but NOT aria-hidden, so it's exposed to screen readers as an unlabeled,
   * meaningless text field. No user ever types into it — exclude it from the accessibility tree
   * entirely, the same treatment already used for WPForms' honeypot field. */
  document.querySelectorAll('.wpforms-recaptcha-hidden').forEach(function (el) {
    el.setAttribute('aria-hidden', 'true');
    el.setAttribute('tabindex', '-1');
  });

  /* axe 'region' (best-practice): Sydney renders its skip-link (<a class="skip-link"> "Skip to
   * content") before the header, outside every landmark, so axe reports "page content not
   * contained by landmarks". Wrapping the skip-link in a labelled <nav> is the only reliable fix —
   * tabindex/role on the skip target, or re-pointing the href, do NOT satisfy axe's region rule
   * (verified empirically). Guarded + idempotent so it never double-wraps. */
  document.querySelectorAll('a.skip-link').forEach(function (sk) {
    if (sk.closest('nav, [role="navigation"]')) return;
    var nav = document.createElement('nav');
    nav.setAttribute('aria-label', 'Skip links');
    sk.parentNode.insertBefore(nav, sk);
    nav.appendChild(sk);
  });

  /* NOTE: former fixes #3 (aria-label on role-less .player-panel-row div) and #4 (h3->h2 heading
   * order) USED to be applied server-side by a companion snippet — but Seriously Simple Podcasting
   * 3.16.3 fixes both natively (role="group" on the div; configurable heading level defaulting to
   * h2), so that snippet was removed on 2026-07-15. Nothing to do here for #3/#4 anymore. */
})();
JS;
	echo "\n<script id=\"a11y-fixes\">" . $js . "</script>\n";
}, 99 );
